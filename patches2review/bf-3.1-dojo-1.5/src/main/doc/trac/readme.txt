Trac Configuration Files:

site.html:
	- template to add header and footer to every trac page
	- has to be placed at $PATH_TO_TRAC/templates/site.html

style.css:
	- default css stylesheet for betterFORM trac
	- has to be placed at $PATH_TO_TRAC/htdocs/style.css

trac.ini:
	- trac configuration file
	- has to be placed at $PATH_TO_TRAC/conf/trac.ini
